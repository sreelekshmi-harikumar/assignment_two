        scalers[col] = scaler
    return scaled, scalers

Signal Processing — STFT & Spectrograms

Apply Short-Time Fourier Transform with a sliding window to convert 1-D financial signals into 2-D time–frequency spectrograms.

Short-Time Fourier Transform
STFT(t, f) = ∫ X(τ) · w(τ − t) · e^{−j2πfτ} dτ

Spectrogram: S(t, f) = |STFT(t, f)|² S ∈ ℝ^{T × F}
Sliding Window

Parameters Used

Window Length L 12 months
Hop Size H 1 month
Overlap 11 months
Window Type Hann
Frequency Interpretation

What Frequencies Mean

  • Low freq (≈0) — multi-year trends, macro cycles, bull/bear regimes
  • Mid freq — quarterly earnings seasonality, annual cycles
  • High freq (≈0.5) — month-to-month noise, one-off shocks, micro-volatility

Large L (e.g. 24 months) Coarser — fewer frames Finer — more freq bins Better at detecting long-term cycles but misses sudden market events
Small L (e.g. 4 months) Finer — more frames Coarser — fewer bins Captures flash crashes precisely but can't resolve seasonal frequency
Large H (large step) Less overlap, faster No effect Efficient but may alias fast-moving events between frames
Small H (H=1) Maximally smooth No effect Most informative spectrograms; chosen for this assignment
Heisenberg Uncertainty (Signal Theory)
Δt · Δf ≥ 1/(2π)
Better frequency resolution always costs time resolution — and vice versa.
task2_signal_processing.py — STFT core 
from scipy.signal import stft

def compute_stft(signal, fs=1.0, nperseg=12, noverlap=11):
    """
    Compute STFT and return spectrogram S ∈ R^{T×F}.
    fs=1.0  →  1 sample per month
    """
    f, t, Zxx = stft(signal, fs=fs, window="hann",
                     nperseg=nperseg, noverlap=noverlap)

    S_freq_time = np.abs(Zxx) ** 2   # energy (F, T)
    S = S_freq_time.T              # transpose → (T, F)
    return f, t, Zxx, S

# Build per-channel spectrogram for CNN input ─────────────────
def build_spectrogram_tensor(segment):    # segment: (L, C)
    specs = []
    for c in range(segment.shape[1]):
        _, _, Zxx = stft(segment[:, c], fs=1.0,
                         nperseg=12, noverlap=11)
        specs.append(np.abs(Zxx) ** 2)     # (F, T_s)
    return np.stack(specs, axis=-1)       # (F, T_s, C)

class SpectrogramCNN(nn.Module):
    def __init__(self, in_channels=5):
        super().__init__()
        def conv_block(i, o, pool=True):
            layers = [nn.Conv2d(i, o, 3, padding=1),
                      nn.BatchNorm2d(o), nn.ReLU(True)]
            if pool: layers.append(nn.MaxPool2d(2,2))
            return nn.Sequential(*layers)

        self.features = nn.Sequential(
            conv_block(in_channels, 32),
            conv_block(32, 64),
            conv_block(64, 128, pool=False),
            nn.AdaptiveAvgPool2d((4, 4)),
        )
        self.regressor = nn.Sequential(
            nn.Flatten(),
            nn.Linear(128*4*4, 64), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(64, 16),  nn.ReLU(),
            nn.Linear(16, 1),
        )

    def forward(self, x):
        return self.regressor(self.features(x)).squeeze(1)

# Training loop ───────────────────────────────────────────────
criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=1e-3, weight_decay=1e-4)
